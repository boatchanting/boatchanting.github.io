三、程序改错题

每题含 4 行错误，请写出修改后的完整代码行。

1. 计算10个数的平均值

#include <iostream>
using namespace std;
int main(){
    int arr[10];
    int sum=0;
    for(int i=1;i<=10;i++){
        cin >> arr[i];
        sum += arr[i];
    }
    double avg = sum / 10;
    cout << "Average: " << avg;
    return 0;
}

2. 输出3x3矩阵对角线和

#include <iostream>
using namespace std;
int main(){
    int mat[3][3] = {1,2,3,4,5,6,7,8,9};
    int sum=0;
    for(int i=0;i<3;i++){
        sum += mat[i][i];
        sum += mat[i][3-i];
    }
    cout << "Sum: " << sum;
    return 0;
}
