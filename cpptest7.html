1. 最小素数 (≥ m)

#include <iostream>
#include <cmath>
using namespace std;
int main(){
    int m,n,k;
    cin >> m;
    for(n=m;;n++){
        bool isPrime=true;
        for(k=2;k<=sqrt(n);k++)      // ①
            if(n%k==0){              // ②
                isPrime=false;break;
            }
        if(isPrime) break;           // ③
    }
    cout << n << endl;
    return 0;                        // ④
}

2. 单链表长度

#include <iostream>
using namespace std;
struct Node{
    int data;
    Node *next;
};
int Length(Node *head){
    int cnt=0;
    for(Node *p=head;p!=NULL;p=p->next)
        cnt++;                       // ①
    return cnt;                      // ②
}
int main(){
    Node *head=NULL;
    cout << Length(head) << endl;   // ③
    return 0;                        // ④
}
