每题含 4 行错误，请写出修改后的完整代码行。

1. 输出Fibonacci数列前20项

#include <iostream>
using namespace std;
int main() {
    int f1=1, f2=1;
    cout << f1 << " " << f2 << " ";
    for(int i=3; i<=20; i++) {
        int f3 = f1 - f2; // 错误1
        cout << f3 << " ";
        f1 = f2;
        f2 = f3;
    }
    return 0;
}

2. 计算圆柱体体积（半径r，高h）

#include <iostream>
#define PI 3.14
using namespace std;
int main() {
    float r, h;
    cin >> r >> h;
    float volume = 2 * PI * r * h; // 错误2
    cout << "Volume: " << volume << endl;
    return 0;
}
