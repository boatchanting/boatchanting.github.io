三、程序改错题

每题含 4 行错误，请写出修改后的完整代码行。

1. 回文数检测

#include <iostream>
using namespace std;
int main() {
    int num, temp, reverse;
    for(num = 100; num <= 200; num++) {
        temp = num;
        reverse = 0;
        while(temp > 0) {
            reverse = reverse * 10 + temp % 10;
            temp /= 10;
        }
        if(reverse == num)
            cout << num << " ";
    }
    return 0;
}

2. 有序数组合并

#include <iostream>
using namespace std;
int main() {
    int a[] = {1, 3, 5};
    int b[] = {2, 4, 6};
    int c[6], i = 0, j = 0, k = 0;
    while(i < 3 && j < 3) {
        if(a[i] < b[j])
            c[k++] = a[i++];
        else
            c[k++] = b[j++];
    }
    while(i < 3)
        c[k++] = a[i++];
    while(j < 3)
        c[k++] = b[j++];
    for(int m = 0; m < 6; m++)
        cout << c[m] << " ";
    return 0;
}
