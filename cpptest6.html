1. 阿姆斯特朗数检测

以下程序试图输出1000以内的所有阿姆斯特朗数（各位数字立方和等于本身的数），但包含4处错误 
#include <iostream>
using namespace std;
int main() {
    int num, temp, digit, sum;
    for(num = 1; num <= 1000; num++) {
        temp = num;
        sum = 0;
        while(temp != 0) {
            digit = temp % 10;
            sum += digit * digit * digit;
            temp /= 10;
        }
        if(sum = num)
            cout << num << " ";
    }
    return 0;
}

2. 奇偶分组

以下程序试图将数组a中的元素按奇偶分组（奇数在前，偶数在后），但包含4处错误 
#include <iostream>
using namespace std;
int main() {
    int a[] = {1, 2, 3, 4, 5, 6};
    int n = sizeof(a) / sizeof(a[0]);
    int left = 0, right = n - 1;
    while(left < right) {
        while(a[left] % 2 == 1) left++;
        while(a[right] % 2 == 0) right--;
        if(left < right) {
            int temp = a[left];
            a[left] = a[right];
            a[right] = temp;
        }
    }
    for(int i = 0; i < n; i++)
        cout << a[i] << " ";
    return 0;
}
