1. 偶数平均值

#include <iostream>
using namespace std;
int main(){
    int arr[5]={2,3,4,5,6};
    int sum=0,count=0;
    for(int i=0;i<=5;i++){      
        if(arr[i] % 2 = 0){      
            sum += arr(i);      
            count++;
        }
    }
    double avg = sum / count;   
    cout << "Average:" << avg << endl;
    return 0;
}

2. 字符串连接

#include <iostream>
using namespace std;
int main(){
    char s1[12]="Hello";
    char s2[]="World";
    int i=0,j=0;
    while(s1[i] != '\0')
        i++;
    while(s2[j] != "\0"){      
        s1[i + j] = s2[j];       
        j++                      
    }
    s1[i + j] = '\0'            
    cout << s1 << endl;
    return 0;
}
